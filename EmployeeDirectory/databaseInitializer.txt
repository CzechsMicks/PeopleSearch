CREATE TABLE EmployeeDirectory  
(  
    ID int NOT NULL  
    ,FName varchar(50) NULL  
    ,LName varchar(50) NULL  
    ,Address varchar(50) NULL  
    ,Age int NULL  
    ,Interests varchar(50) NULL  
    ,Picture varchar(50) NULL
	,GUID uniqueidentifier NOT NULL 
	,PRIMARY KEY (ID)
);
